import pdfplumber
import pandas as pd
import os

def extract_from_pdf(pdf_path):
    
    all_text = ""
    tables = []
    
    # Detect PDF type based on filename
    filename = os.path.basename(pdf_path).upper()
    if "USM" in filename:
        pdf_type = "USM"
    elif "LPO" in filename:
        pdf_type = "LPO"
    else:
        pdf_type = "DEFAULT"
    
    print(f"📄 Detected PDF type: {pdf_type}")

    try:
        with pdfplumber.open(pdf_path) as pdf:
            for page_num, page in enumerate(pdf.pages, start=1):
                print(f"Processing page {page_num}...")
                text = page.extract_text()
                if text:
                    all_text += text + "\n\n"

                page_tables = page.extract_tables()
                for table in page_tables:
                    if table and len(table) > 1:  # Ensure table has data rows
                        # For LPO files, we need better table handling
                        if pdf_type == "LPO":
                            # Try to split the table if it's all in one column
                            df = extract_lpo_table(table)
                            if df is not None:
                                tables.append(df)
                        else:
                            # Original USM table handling
                            df = pd.DataFrame(table[1:], columns=table[0])
                            tables.append(df)

        print(f"✓ Extracted {len(all_text)} characters and {len(tables)} tables")
        return all_text, tables, pdf_type

    except Exception as e:
        print(f"Error extracting PDF: {e}")
        return "", [], pdf_type

def extract_lpo_table(table):
    try:
        if not table or len(table) < 2:
            return None
        
        print(f"Raw table has {len(table)} rows")
        
        # Check if this is a line items table (has "Bil Keterangan" in header)
        is_line_items_table = False
        for row in table:
            if row and any(keyword in str(row).upper() for keyword in ["BIL KETERANGAN", "BIL", "KETERANGAN"]):
                is_line_items_table = True
                break
        
        if not is_line_items_table:
            print("Not a line items table, skipping")
            return None
        
        # Find the header row
        header_row = 0
        for i, row in enumerate(table):
            if row and any(keyword in str(row).upper() for keyword in ["BIL", "KETERANGAN", "KUANTITI"]):
                header_row = i
                print(f"Found header at row {i}: {row}")
                break
        
        if header_row >= len(table) - 1:
            print("No valid header found")
            return None
            
        # Extract header and data rows
        headers = table[header_row]
        data_rows = table[header_row + 1:]
        
        print(f"Headers: {headers}")
        print(f"Data rows: {len(data_rows)}")
        
        # Check if we have a single column with all data
        if len(headers) == 1 and headers[0] and "BIL KETERANGAN" in str(headers[0]).upper():
            print("Detected single column table, parsing line items...")
            return parse_single_column_table(data_rows)
        
        # Clean headers for multi-column table
        clean_headers = []
        for header in headers:
            if header:
                clean_headers.append(str(header).strip())
            else:
                clean_headers.append(f"Column_{len(clean_headers)}")
        
        # Create DataFrame
        df = pd.DataFrame(data_rows, columns=clean_headers)
        
        # Remove empty rows
        df = df.dropna(how='all')
        
        # Clean data in each column
        for col in df.columns:
            df[col] = df[col].astype(str).str.strip()
            df[col] = df[col].replace('None', '')
            df[col] = df[col].replace('nan', '')
        
        print(f"Final DataFrame shape: {df.shape}")
        return df
        
    except Exception as e:
        print(f"Error processing LPO table: {e}")
        return None

def parse_single_column_table(data_rows):
    try:
        line_items = []
        
        for row in data_rows:
            if not row or not row[0]:
                continue
                
            # Get the text content
            text = str(row[0]).strip()
            
            # Skip empty or header-like rows
            if not text or text == 'nan' or any(header_word in text.upper() for header_word in ['BIL', 'KETERANGAN', 'JUMLAH', 'TOTAL']):
                continue
            
            # Try to parse the line item
            # Pattern: number code quantity price total
            import re
            
            # Look for patterns like "1 D07.0017.05 30.00 71.77 2,153.10"
            pattern = r'^(\d+)\s+([A-Z0-9\.]+)\s+(\d+\.?\d*)\s+(\d+\.?\d*)\s+([\d,]+\.?\d*)$'
            match = re.match(pattern, text)
            
            if match:
                bil, code, qty, price, total = match.groups()
                line_items.append({
                    'Bil': bil,
                    'Keterangan': code,
                    'Kuantiti': qty,
                    'Harga Seunit': price,
                    'JUMLAH': total
                })
                print(f"Parsed line item: {bil} - {code}")
            else:
                # Try to extract just the description if it doesn't match the pattern
                if len(text) > 10:  # Only consider substantial text
                    line_items.append({
                        'Bil': '',
                        'Keterangan': text,
                        'Kuantiti': '',
                        'Harga Seunit': '',
                        'JUMLAH': ''
                    })
                    print(f"Added description: {text[:50]}...")
        
        if line_items:
            df = pd.DataFrame(line_items)
            print(f"Parsed {len(line_items)} line items from single column")
            return df
        else:
            print("No line items parsed from single column")
            return None
            
    except Exception as e:
        print(f"Error parsing single column table: {e}")
        return None

import re

def clean_text(text):
    if not text:
        return ""
    
    # Replace newlines with spaces
    text = text.replace("\n", " ")

    # Replace multiple spaces with a single space
    text = re.sub(r"\s+", " ", text)

    # Remove leading and trailing whitespace
    text = text.strip()

    return text

def clean_text_preserve_lines(text):
    if not text:
        return ""

    lines = text.split("\n")
    cleaned_lines = []

    for line in lines:
        # Clean each line but keep it as a separate line
        cleaned_line = re.sub(r"\s+", " ", line).strip()
        if cleaned_line:  # Only keep non-empty lines
            cleaned_lines.append(cleaned_line)

    return "\n".join(cleaned_lines)

import re

def classify_by_rule(text, pdf_type="DEFAULT"):
    sections = {
        "header": [],
        "supplier_info": [],
        "item_table": [],
        "approval": [],
        "terms_conditions": [],
        "other": []
    }

    for line in text.split("\n"):
        line = line.strip()
        if not line:
            continue

        # Check for header keywords
        if any(keyword in line.upper() for keyword in ["PURCHASE ORDER", "PESANAN RASMI", "PO NUMBER", "LPO"]):
            sections["header"].append(line)

        # Check for supplier information
        elif any(keyword in line.upper() for keyword in ["PHARMANIAGA", "SUPPLIER", "ALAMAT PENGHANTARAN", "DELIVERY ADDRESS"]):
            sections["supplier_info"].append(line)

        # Check for item table headers
        elif any(keyword in line.upper() for keyword in ["BIL.", "ITEM", "KETERANGAN", "DESCRIPTION", "QUANTITY", "KUANTITI"]):
            sections["item_table"].append(line)

        # Check for approval section
        elif any(keyword in line.upper() for keyword in ["DILULUSKAN", "APPROVED BY", "AUTHORIZED", "DILULUSKAN OLEH"]):
            sections["approval"].append(line)

        # Check for terms and conditions
        elif any(keyword in line.upper() for keyword in ["TERMS AND CONDITIONS", "TERMS & CONDITIONS", "T&C"]):
            sections["terms_conditions"].append(line)

        # Everything else goes to "other"
        else:
            sections["other"].append(line)

    return sections

def extract_lpo_info(text, tables):
    import re
    
    lpo_info = {
        'po_number': '',
        'order_date': '',
        'ship_to': '',
        'contact_person': '',
        'delivery_date': '',
        'line_items': []
    }
    
    try:
        lines = text.split('\n')
        
        for line in lines:
            if 'LPO' in line.upper() and any(char.isdigit() for char in line):
                # Clean up the line to get just the PO number
                po_match = line.strip()
                if 'LPO/' in po_match:
                    lpo_info['po_number'] = po_match
                break
        
        date_pattern = r'\d{2}/\d{2}/\d{4}'
        for line in lines:
            if re.search(date_pattern, line):
                # Check if this line contains date-related keywords or LPO number
                if any(keyword in line.upper() for keyword in ['TARIKH', 'DATE', 'PESANAN', 'LPO/']):
                    # Extract just the date part
                    date_match = re.search(date_pattern, line)
                    if date_match:
                        lpo_info['order_date'] = date_match.group()
                        print(f"Found order date: {date_match.group()}")
                        break
        
        # Extract ship to address - look for delivery address at the beginning of document
        # The delivery address is typically at the top, before contact info and order details
        for i, line in enumerate(lines):
            # Look for lines that contain address-like patterns (numbers, street names, postal codes)
            # and are in the first 10 lines (before order details start)
            if i < 10 and line.strip():
                # Check if this line looks like an address (contains numbers, common address words)
                address_indicators = ['LORONG', 'JALAN', 'PERSIARAN', 'KAMPUNG', 'TAMAN', 'BANDAR', 
                                    'SELANGOR', 'KUALA LUMPUR', 'JOHOR', 'PENANG', 'PERAK', 'KEDAH',
                                    'HOSPITAL', 'KLINIK', 'PUSAT', 'KOMPLEKS', 'BANGUNAN']
                
                # Check if line contains address indicators or postal code pattern
                has_address_indicators = any(indicator in line.upper() for indicator in address_indicators)
                has_postal_code = re.search(r'\d{5}', line)  # 5-digit postal code
                
                if has_address_indicators or has_postal_code:
                    # Look forward to get the complete address
                    address_lines = [line.strip()]
                    j = i + 1
                    # Look at the next few lines for the complete address
                    while j < len(lines) and j < i + 5:  # Look at next 5 lines max
                        if lines[j].strip() and not any(keyword in lines[j].upper() for keyword in ['T 603', 'E hpupm', 'PESANAN BELIAN', 'TARIKH', 'NO PESANAN']):
                            address_lines.append(lines[j].strip())
                        j += 1
                    
                    if len(address_lines) >= 2:  # At least 2 lines to be a valid address
                        lpo_info['ship_to'] = ' '.join(address_lines)
                        print(f"Found ship to address (delivery address): {lpo_info['ship_to']}")
                        break
        
        # Extract contact person - look for specific pattern
        for line in lines:
            if 'CONTACT PERSON' in line.upper() or 'KONTAK' in line.upper():
                # Extract name after the colon
                if ':' in line:
                    name_part = line.split(':')[-1].strip()
                    lpo_info['contact_person'] = name_part
                else:
                    lpo_info['contact_person'] = line.strip()
                break
        
        # Extract delivery date
        for line in lines:
            if 'PADA ATAU SEBELUM' in line.upper() or 'DELIVERY' in line.upper():
                # Extract date from this line
                date_match = re.search(date_pattern, line)
                if date_match:
                    lpo_info['delivery_date'] = date_match.group()
                else:
                    lpo_info['delivery_date'] = line.strip()
                break
        
        # Extract line items from text - improved extraction
        # The line items are actually in the text, not in tables
        print("Extracting line items from text...")
        
        # Look for line item patterns in the text
        lines = text.split('\n')
        
        for i, line in enumerate(lines):
            # Look for lines that start with a number and contain a product code
            if re.match(r'^\d+\s+[A-Z0-9\.]+\(cid:\d+\)', line.strip()):
                # This looks like a line item header
                # Extract: number + code + quantity + price + total
                parts = line.strip().split()
                if len(parts) >= 5:
                    bil = parts[0]
                    code = parts[1]
                    qty = parts[2]
                    price = parts[3]
                    total = parts[4]
                    
                    # Get the description from the next line(s)
                    description = ""
                    j = i + 1
                    while j < len(lines) and lines[j].strip() and not re.match(r'^\d+\s+[A-Z0-9\.]+\(cid:\d+\)', lines[j].strip()):
                        if lines[j].strip():
                            description += lines[j].strip() + " "
                        j += 1
                    
                    item = {
                        'description': description.strip() or f"{code} - Line item {bil}",
                        'quantity': qty,
                        'unit_price': price,
                        'total': total
                    }
                    lpo_info['line_items'].append(item)
                    print(f"Found line item: {bil} - {code} - Qty: {qty} - Price: {price} - Total: {total}")
                    print(f"Description: {description.strip()}")
        
        # Also try to extract from tables as backup
        for table in tables:
            if table is not None and not table.empty:
                print(f"Processing table with {len(table)} rows and columns: {list(table.columns)}")
                
                # Look for columns that match our expected structure
                columns = [col.upper() for col in table.columns]
                
                # Find the right columns
                desc_col = None
                qty_col = None
                price_col = None
                total_col = None
                
                for i, col in enumerate(columns):
                    if 'KETERANGAN' in col or 'DESCRIPTION' in col:
                        desc_col = table.columns[i]
                    elif 'KUANTITI' in col or 'QUANTITY' in col:
                        qty_col = table.columns[i]
                    elif 'HARGA' in col or 'PRICE' in col:
                        price_col = table.columns[i]
                    elif 'JUMLAH' in col or 'TOTAL' in col:
                        total_col = table.columns[i]
                
                print(f"Found columns - Desc: {desc_col}, Qty: {qty_col}, Price: {price_col}, Total: {total_col}")
                
                # Extract line items - improved logic
                for idx, row in table.iterrows():
                    # Check if this row has meaningful data
                    has_data = False
                    for col in table.columns:
                        cell_value = str(row[col]).strip()
                        if cell_value and cell_value != 'nan' and cell_value != 'None' and len(cell_value) > 1:
                            has_data = True
                            break
                    
                    if has_data:
                        # Skip header rows
                        if desc_col and any(header_word in str(row[desc_col]).upper() for header_word in ['BIL', 'KETERANGAN', 'DESCRIPTION']):
                            continue
                        
                        # Skip total/summary rows
                        if desc_col and any(summary_word in str(row[desc_col]).upper() for summary_word in ['JUMLAH', 'TOTAL', 'RINGGIT']):
                            continue
                        
                        item = {
                            'description': str(row[desc_col]).strip() if desc_col and str(row[desc_col]).strip() != 'nan' else '',
                            'quantity': str(row[qty_col]).strip() if qty_col and str(row[qty_col]).strip() != 'nan' else '',
                            'unit_price': str(row[price_col]).strip() if price_col and str(row[price_col]).strip() != 'nan' else '',
                            'total': str(row[total_col]).strip() if total_col and str(row[total_col]).strip() != 'nan' else ''
                        }
                        
                        # Only add if we have a meaningful description
                        if item['description'] and len(item['description']) > 2:
                            lpo_info['line_items'].append(item)
                            print(f"Added item from table: {item['description'][:50]}...")
        
        print(f"✓ Extracted LPO info: {len(lpo_info['line_items'])} line items")
        return lpo_info
        
    except Exception as e:
        print(f"Error extracting LPO info: {e}")
        return lpo_info

import os

def get_connection():
    try:
        conn = mysql.connector.connect(
            host=os.getenv("DB_HOST", "localhost"),
            user=os.getenv("DB_USER", "root"),
            password=os.getenv("DB_PASSWORD", ""),
            database=os.getenv("DB_NAME", "pdf_classifier")
        )
        return conn
    except mysql.connector.Error as e:
        print(f"Error connecting to database: {e}")
        return None

def init_db():
    try:
        conn = get_connection()
        if not conn:
            print("❌ Cannot connect to database!")
            return False
            
        cursor = conn.cursor()
        
        # Create database if it doesn't exist
        cursor.execute("CREATE DATABASE IF NOT EXISTS pdf_classifier")
        cursor.execute("USE pdf_classifier")
        
        # Create classified_texts table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS classified_texts (
                id INT AUTO_INCREMENT PRIMARY KEY,
                text_line TEXT NOT NULL,
                category VARCHAR(100) NOT NULL,
                pdf_type VARCHAR(50) DEFAULT 'DEFAULT',
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        # Create lpo_data table for LPO exports
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS lpo_data (
                id INT AUTO_INCREMENT PRIMARY KEY,
                po_number VARCHAR(100),
                order_date VARCHAR(100),
                ship_to TEXT,
                contact_person VARCHAR(200),
                delivery_date VARCHAR(100),
                description TEXT,
                quantity VARCHAR(50),
                unit_price VARCHAR(50),
                total_amount VARCHAR(50),
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        conn.commit()
        cursor.close()
        conn.close()
        
        print("✅ Database initialized successfully!")
        return True
        
    except Exception as e:
        print(f"Error initializing database: {e}")
        return False

"""
Main Program - Enhanced PDF Text Classifier
Processes both USM and LPO PDF files with filename detection
"""
from modules.pdf_extractor import extract_from_pdf
from modules.text_cleaner import clean_text_preserve_lines
from modules.rule_classifier import classify_by_rule, display_classification, extract_lpo_info, display_lpo_info
from modules.excel_exporter import export_lpo_to_excel
from modules.db_manager import init_db, save_lpo_to_db, save_to_db
import pandas as pd

def main():
    """Main program workflow with support for both USM and LPO PDFs."""
    print("="*60)
    print(" ENHANCED PDF TEXT CLASSIFIER")
    print(" Supports both USM and LPO PDF types")
    print("="*60)
    
    # Step 0: Initialize database
    print("\n[0/5] Initializing database...")
    if not init_db():
        print("⚠️  Database initialization failed. Continuing without database...")
        use_db = False
    else:
        use_db = True
    
    # Step 1: Choose PDF file
    print("\n[1/5] Available PDF files:")
    print("1. USMPO_25_36690.pdf (USM format)")
    print("2. LPO-1001-002938-2025..pdf (LPO format)")
    
    choice = input("\nEnter your choice (1 or 2): ").strip()
    
    if choice == "1":
        pdf_path = "data/USMPO_25_36690.pdf"
    elif choice == "2":
        pdf_path = "data/LPO-1001-002938-2025..pdf"
    else:
        print("Invalid choice. Using LPO file as default.")
        pdf_path = "data/LPO-1001-002938-2025..pdf"
    
    print(f"\n[2/5] Processing: {pdf_path}")
    
    # Step 2: Extract text and tables
    print(f"\n[3/5] Extracting text and tables...")
    text, tables, pdf_type = extract_from_pdf(pdf_path)
    
    if not text:
        print("❌ No text extracted from PDF!")
        return
    
    # Step 3: Clean text
    print(f"\n[4/5] Cleaning text...")
    cleaned_text = clean_text_preserve_lines(text)
    
    # Step 4: Process based on PDF type
    print(f"\n[5/5] Processing {pdf_type} format...")
    
    if pdf_type == "LPO":
        # Extract LPO information
        lpo_info = extract_lpo_info(cleaned_text, tables)
        
        # Display LPO information
        display_lpo_info(lpo_info)
        
        # Export to Excel
        if lpo_info['line_items']:
            output_file = "lpo_exported_data.xlsx"
            export_lpo_to_excel(lpo_info, output_file)
            print(f"\n✅ LPO data exported to: {output_file}")
            
            # Save to database if available
            if use_db:
                print("\n💾 Saving LPO data to database...")
                save_lpo_to_db(lpo_info)
        else:
            print("\n❌ No line items found to export!")
            
    else:
        # Use original USM classification
        rules_result = classify_by_rule(cleaned_text, pdf_type)
        display_classification(rules_result)
        
        # Save to database if available
        if use_db:
            print("\n💾 Saving classified text to database...")
            # Convert classification results to DataFrame
            all_lines = []
            for category, lines in rules_result.items():
                for line in lines:
                    all_lines.append({"text": line, "class": category})
            
            if all_lines:
                df = pd.DataFrame(all_lines)
                save_to_db(df, pdf_type)
    
    print("\n" + "="*60)
    print(" PROCESSING COMPLETE!")
    print("="*60)

if __name__ == "__main__":
    main()
